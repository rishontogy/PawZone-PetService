# ⚙️ 11. AUTO POINT DETECTION FUNCTION

---

```js
function findMatchingPoints(
  sellerPoints,
  buyerPoints,
  transporterStops
) {
  const pickup = sellerPoints.find(p =>
    transporterStops.includes(p)
  );

  const delivery = buyerPoints.find(p =>
    transporterStops.includes(p)
  );

  if (pickup && delivery) {
    return {
      valid: true,
      pickup,
      delivery
    };
  }

  return { valid: false };
}
```
