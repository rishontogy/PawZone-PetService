# ✅ 2. USE FILE INPUT (REMOVE URL SYSTEM)

## Replace any URL input with:

```html
<input type="file" accept="video/*" id="videoUpload" />
```

---

---

# 🎥 3. READ VIDEO FILE PROPERLY

---

```js
const videoInput = document.getElementById("videoUpload");

videoInput.addEventListener("change", function () {
  const file = videoInput.files[0];

  if (!file) return;

  // Limit size (IMPORTANT)
  if (file.size > 20 * 1024 * 1024) {
    return alert("Video must be under 20MB");
  }

  const reader = new FileReader();

  reader.onload = function (e) {
    const base64Video = e.target.result;

    localStorage.setItem("temp_video", base64Video);

    document.getElementById("videoPreview").src = base64Video;
  };

  reader.readAsDataURL(file);
});
```
