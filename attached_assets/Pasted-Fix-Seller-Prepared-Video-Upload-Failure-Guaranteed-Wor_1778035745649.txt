# ✅ 3. FILE READ HANDLER (FIXED)

---

```js
document
  .getElementById("preparedVideoInput")
  .addEventListener("change", function (e) {
    const file = e.target.files[0];

    if (!file) {
      alert("No file selected");
      return;
    }

    // Size check (20MB max)
    if (file.size > 20 * 1024 * 1024) {
      alert("Video too large (Max 20MB)");
      return;
    }

    const reader = new FileReader();

    reader.onload = function (event) {
      preparedVideoData = event.target.result;

      document.getElementById("preparedVideoPreview").src =
        preparedVideoData;

      console.log("Video loaded successfully");
    };

    reader.onerror = function () {
      alert("Failed to read video file");
    };

    reader.readAsDataURL(file);
  });
```

# ✅ 4. UPLOAD FUNCTION (CRITICAL FIX)

---

```js
function handlePreparedVideoUpload() {
  if (!preparedVideoData) {
    alert("Please select a video first");
    return;
  }

  const orderId = currentOrder.id;

  const orders = JSON.parse(localStorage.getItem("orders")) || [];

  const index = orders.findIndex(o => o.id === orderId);

  if (index === -1) {
    alert("Order not found");
    return;
  }

  orders[index].prepared_video = preparedVideoData;

  localStorage.setItem("orders", JSON.stringify(orders));

  alert("Prepared video uploaded successfully");

  // Clear temp
  preparedVideoData = null;

  document.getElementById("preparedVideoInput").value = "";
}
```
