```js
function calculateOrder(order) {
  const itemTotal = Number(order.item_total || 0);
  const transport = Number(order.transport_charge || 0);

  const buyerFee = itemTotal > 100 ? 20 : 5;
  const sellerFee = buyerFee;
  const transporterFee = transport > 200 ? 40 : 20;

  const buyerTotal = itemTotal + buyerFee + transport;

  const sellerPayout = itemTotal - sellerFee;
  const transporterPayout = transport - transporterFee;

  const platformFee = buyerFee + sellerFee + transporterFee;

  return {
    buyerTotal,
    sellerPayout,
    transporterPayout,
    platformFee,
    buyerFee,
    sellerFee,
    transporterFee
  };
}
```
