## Replace existing logic with:

```js id="correct_pf"
function calculatePlatformFee(order) {
  const itemTotal = Number(order.item_total || 0);
  const transport = Number(order.transport_charge || 0);

  const buyerFee = itemTotal > 100 ? 20 : 5;
  const sellerFee = buyerFee;

  const transportFee = transport > 200 ? 40 : 20;

  const salePlatformFee = buyerFee + sellerFee;
  const transportPlatformFee = transportFee;

  const totalPlatformFee = salePlatformFee + transportPlatformFee;

  return {
    salePlatformFee,
    transportPlatformFee,
    totalPlatformFee
  };
}
```
