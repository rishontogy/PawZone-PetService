# 🎥 2. DELIVERY VIDEO UPLOAD FIX (CRITICAL)

---

## ❌ Problem:

* Upload failing due to wrong API / JSON error

---

## ✅ FIX: USE LOCAL FILE HANDLING (NO API)

---

### HTML:

```html
<input type="file" id="deliveryVideoInput" accept="video/*" />
<video id="deliveryPreview" controls width="200"></video>
<button onclick="uploadDeliveryVideo()">Upload Delivery Video</button>
```

---

---

### JS (WORKING VERSION):

```js
let deliveryVideoData = null;

document.getElementById("deliveryVideoInput")
.addEventListener("change", function (e) {
  const file = e.target.files[0];

  if (!file) return;

  if (file.size > 20 * 1024 * 1024) {
    alert("Max 20MB video allowed");
    return;
  }

  const reader = new FileReader();

  reader.onload = function (event) {
    deliveryVideoData = event.target.result;

    document.getElementById("deliveryPreview").src = deliveryVideoData;
  };

  reader.readAsDataURL(file);
});
```

---

### SAVE FUNCTION:

```js
function uploadDeliveryVideo() {
  if (!deliveryVideoData) {
    alert("Upload video first");
    return;
  }

  const orders = JSON.parse(localStorage.getItem("orders")) || [];
  const index = orders.findIndex(o => o.id === currentOrder.id);

  orders[index].delivery_video = deliveryVideoData;
  orders[index].status = "DELIVERED_PENDING_CONFIRMATION";

  localStorage.setItem("orders", JSON.stringify(orders));

  alert("Delivery video uploaded");
}
```
